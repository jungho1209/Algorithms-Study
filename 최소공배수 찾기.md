# 최소공배수

## 최소공배수 = 두 수의 곱 / 최대공약수

```c
#include <stdio.h>
int lcm(int a, int b);
int main() {
	int a, b, wl;

	scanf("%d", &wl);
	while (wl--) {
		scanf("%d %d", &a, &b);
		printf("%d\n", lcm(a, b));
	}
	return 0;
}
int lcm(int a, int b) {
	int max = 0;
	int sum = a * b;
	int maxt = 0;
	int c;
	if (a > b) {
		c = b;
	}
	else {
		c = a;
	}
	for (int i = 1; i <= c; i++) {
		if (a % i == 0 && b % i == 0) {
			maxt = i;
		}
	}
	int end = sum / maxt;
	return end;
```
